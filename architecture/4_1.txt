

module counter10 (
    input  wire CLK, 
    input  wire RST,
    output reg [3:0] Q
);

always @(posedge CLK) begin
        if (RST) begin
            Q <= 4'd0;
        end else if (Q == 4'd9) begin
            Q <= 4'd0;
        end else begin
            Q <= Q + 4'd1;
        end
    end

endmodule
