単体法ソルバーの実装例 (simplex.py)
import numpy as np

def simplex(c, A, b, sense="max", verbose=False):
    """
    タブロー単体法による線形計画問題ソルバー
    対象: maximize c^T x  s.t. Ax <= b, x >= 0, b >= 0 [1]
    """
    m, n = A.shape
    # 初期タブローの作成 [5, 6]
    # 行構成: [z行] + [制約m行]
    # 列構成: [決定変数n] + [スラック変数m] + [RHS]
    tableau = np.zeros((m + 1, n + m + 1))
    
    # 制約式の埋め込み (A x + I s = b) [6]
    tableau[1:, :n] = A
    tableau[1:, n:n+m] = np.eye(m)
    tableau[1:, -1] = b
    
    # 目的関数行の埋め込み (z - c^T x = 0) [5]
    # 最大化なら -c, 最小化なら c
    if sense == "max":
        tableau[0, :n] = -np.array(c)
    else:
        tableau[0, :n] = np.array(c)
        
    num_iterations = 0
    # 初期基底はスラック変数 [7, 8]
    basis = list(range(n, n + m))
    
    while True:
        if verbose:
            print(f"--- Iteration {num_iterations} ---")
            print(tableau)
        
        # 1. 入る変数の選択 (Dantzig 規則) [9]
        # z行で最も負の係数を持つ列を選ぶ
        z_row = tableau[0, :-1]
        entering_col = np.argmin(z_row)
        
        # 停止条件: すべての係数が 0 以上になれば最適 [4, 10]
        if z_row[entering_col] >= -1e-9:
            status = "optimal"
            break
            
        # 2. 出る変数の選択 (最小比検定) [11]
        ratios = []
        for i in range(1, m + 1):
            val = tableau[i, entering_col]
            if val > 1e-9:
                ratios.append(tableau[i, -1] / val)
            else:
                ratios.append(np.inf)
        
        leaving_row_idx = np.argmin(ratios)
        leaving_row = leaving_row_idx + 1
        
        # 非有界判定: 増やせる限界がない場合 [10]
        if ratios[leaving_row_idx] == np.inf:
            status = "unbounded"
            return {"status": status}
            
        # 3. ピボット操作 (行基本変形) [2, 3, 12]
        pivot_element = tableau[leaving_row, entering_col]
        num_iterations += 1
        basis[leaving_row_idx] = entering_col # 基底の更新 [2]
        
        # ピボット行をピボット要素で割る [3]
        tableau[leaving_row, :] /= pivot_element
        
        # 他の行から引いて、ピボット列を単位ベクトル化する [3, 12]
        for i in range(m + 1):
            if i != leaving_row:
                factor = tableau[i, entering_col]
                tableau[i, :] -= factor * tableau[leaving_row, :]
                
    # 結果の抽出
    x = np.zeros(n)
    for i, b_idx in enumerate(basis):
        if b_idx < n:
            x[b_idx] = tableau[i + 1, -1]
            
    objective_val = tableau[0, -1]
    # 双対解 y* を最終タブローのz行のスラック変数係数から読み取る [13, 14]
    y = tableau[0, n:n+m]
    
    return {
        "status": status,
        "x": x,
        "objective": objective_val,
        "y": y,
        "num_iterations": num_iterations,
        "basis": basis
    }
実装のポイント（資料との対応）
標準形への変換: Ax <= b を等式にするため、単位行列 I を用いてスラック変数を導入しています
。
Dantzig 規則: z 行の係数の中で最も負の値を持つ列を「入る変数」として選択します

最小比検定: 右辺 (RHS) を「入る変数」の係数で割り、最小の正の比を持つ行を「出る変数」とします

双対解の抽出: 最終タブローの z 行におけるスラック変数の位置にある値が、そのまま双対解（影の価格）となります
